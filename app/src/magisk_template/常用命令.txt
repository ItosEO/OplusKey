# 微信付款码
am start -n com.tencent.mm/.plugin.offline.ui.WalletOfflineCoinPurseUI

# 支付宝付款码
am start -a android.intent.action.VIEW -d "alipays://platformapi/startapp?appId=20000056"

# 支付宝扫码
am start -n com.eg.android.AlipayGphone/com.alipay.mobile.scan.as.main.MainCaptureActivity

# 微信扫码
am start -n com.tencent.mm/.plugin.scanner.ui.BaseScanUI

# 开手电筒（只支持骁龙设备）
echo 500 > /sys/class/leds/led:torch_0/brightness;echo 1 > /sys/class/leds/led:switch_0/brightness

# 关手电筒（只支持骁龙设备）
echo torch0_trigger > /sys/class/leds/led:torch_0/trigger;echo 0 > /sys/class/leds/led:torch_0/brightness;echo switch0_trigger > /sys/class/leds/led:switch_0/trigger;echo torch0_trigger > /sys/class/leds/led:torch_0/trigger

# 开关蓝牙
svc bluetooth enable
svc bluetooth disable

# 开录音
am start --user 0 -n com.coloros.soundrecorder/.TransparentActivity >/dev/null 2>&1

# 关录音
am force-stop com.coloros.soundrecorder

# 开录屏
screenrecord /sdcard/demo.mp4 >/dev/null 2>&1 &
参数：
--size WxH	设置录制分辨率
--bit-rate X	设置比特率 Mbps，默认 4 Mbps
--time-limit	最大录制时长（秒），默认/上限一般为 180 秒
--bugreport	添加时间戳等调试信息

# 关录屏
pkill -l screenrecord

# 打开相机
am start -n com.oplus.camera/com.oplus.camera.Camera

# 开始录制视频
am start -W -a android.media.action.VIDEO_CAMERA && sleep 1 && input keyevent KEYCODE_VOLUME_UP

# 停止录制视频
input keyevent KEYCODE_VOLUME_UP

# 静音
cmd audio set-ringer-mode SILENT
# 震动
cmd audio set-ringer-mode VIBRATE
# 响铃
cmd audio set-ringer-mode NORMAL

# 开旁路充电
echo 1 > /sys/class/oplus_chg/battery/mmi_charging_enable
# 关旁路充电
echo 0 > /sys/class/oplus_chg/battery/mmi_charging_enable

# Clash for Android内测版（v3.xxx）开关
am startservice com.github.kr328.clash/com.github.kr328.clash.VpnService
am stopservice com.github.kr328.clash/com.github.kr328.clash.VpnService
注：只支持常规模式
